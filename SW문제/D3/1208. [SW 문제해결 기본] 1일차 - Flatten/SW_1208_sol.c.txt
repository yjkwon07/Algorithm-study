#include stdio.h
 
int main() {
    for (int tc = 1; tc = 10; tc++) {
         데이터 입력
        int h[1001], mov, tmp;
        int top = -1, bot = 101;
        scanf(%d, &mov);
        for (int i = 0; i  100; i++)
            h[i] = 0;
        for (int i = 0; i  100; i++) {
            scanf(%d, &tmp);
            top = top  tmp  tmp  top;
            bot = bot  tmp  tmp  bot;
            h[tmp]++;
        }
         이동 시작
        for (int i = 0; i  mov && top - bot  1; i++) {
             층 이동
            h[top]--;
            h[top - 1]++;
            h[bot]--;
            h[bot + 1]++;
             top, bot 인덱스 체크
            top = h[top]  0  top  top - 1;
            bot = h[bot]  0  bot  bot + 1;
        }
        printf(#%d %dn, tc, top - bot);
    }
    return 0;
}
